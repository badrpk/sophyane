API reference

# 1) Request OTP (sent from badrpk@gmail.com) — signup once for new users
POST /api/v1/auth/request-otp
{"email":"you@x.com","name":"You","plan":"hybrid","purpose":"signup"}
→ { ok, message: "OTP sent..." }

# 2) Verify OTP → API key issued
POST /api/v1/auth/verify-otp
{"email":"you@x.com","otp":"123456","purpose":"signup"}
→ { api_key, user, auth: "signup" }

# Login (existing user)
POST /api/v1/auth/request-otp {"email":"you@x.com","purpose":"login"}
POST /api/v1/auth/verify-otp  {"email":"you@x.com","otp":"123456","purpose":"login"}

POST /api/v1/chat
Authorization: Bearer sph_...
{"message":"Hello","edge":true}
→ { reply, usage }

GET  /api/v1/usage
Authorization: Bearer sph_...

GET  /api/v1/pricing
GET  /api/v1/health

POST /api/v1/keys   # mint additional keys
Authorization: Bearer sph_...
{"label":"prod"}
